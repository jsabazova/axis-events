<script>
let currentStep = 1;
let countdownInterval1 = null;
let countdownInterval2 = null;
let quoteData = {
quoteId: ‘AX-‘ + Math.floor(1000 + Math.random() * 9000),
show: null,
freightIn: null,
freightOut: null,
pickupAddress: ”,
deliveryAddress: ”,
pickupForklift: null,
deliveryForklift: null
};

// DOM elements
const steps = [‘step1’, ‘step2’, ‘step3’, ‘step4’, ‘step5’, ‘step6’];
const indicators = [
document.getElementById(‘step1-indicator’),
document.getElementById(‘step2-indicator’),
document.getElementById(‘step3-indicator’),
document.getElementById(‘step4-indicator’),
document.getElementById(‘step5-indicator’),
document.getElementById(‘step6-indicator’)
];
const quoteIdSpan = document.getElementById(‘quoteId’);
quoteIdSpan.textContent = ‘Quote ID: #’ + quoteData.quoteId;

function updateStep(step) {
// Clear any existing countdowns
if (countdownInterval1) clearInterval(countdownInterval1);
if (countdownInterval2) clearInterval(countdownInterval2);

currentStep = step;
steps.forEach((s, i) => {
const el = document.getElementById(s);
if (el) el.classList.add(‘hidden’);
});

const currentEl = document.getElementById(steps[step-1]);
if (currentEl) currentEl.classList.remove(‘hidden’);

indicators.forEach((ind, i) => {
ind.classList.remove(‘active’, ‘completed’);
if (i + 1 < step) ind.classList.add(‘completed’);
else if (i + 1 === step) ind.classList.add(‘active’);
});
}

function startCountdown(elementId, seconds, nextStep, callback) {
let timeLeft = seconds;
const messageEl = document.getElementById(elementId);

const interval = setInterval(() => {
timeLeft–;
if (messageEl) {
messageEl.textContent = `Loading next step in ${timeLeft} seconds`;
}
if (timeLeft <= 0) {
clearInterval(interval);
if (callback) callback();
else updateStep(nextStep);
}
}, 1000);

return interval;
}

// Show selection
document.querySelectorAll(‘.show-card’).forEach(card => {
card.addEventListener(‘click’, () => {
document.querySelectorAll(‘.show-card’).forEach(c => c.classList.remove(‘selected’));
card.classList.add(‘selected’);

quoteData.show = {
name: card.querySelector(‘.show-name’).textContent,
venue: card.querySelector(‘.show-venue’).textContent,
dates: card.querySelector(‘.show-dates’).textContent
};

setTimeout(() => updateStep(2), 300);
});
});

// Search functionality
document.getElementById(‘showSearch’).addEventListener(‘input’, (e) => {
const search = e.target.value.toLowerCase();
document.querySelectorAll(‘.show-card’).forEach(card => {
const text = card.textContent.toLowerCase();
card.style.display = text.includes(search) ? ‘block’ : ‘none’;
});
});

// Freight In
document.getElementById(‘freightInYes’).addEventListener(‘click’, () => {
quoteData.freightIn = true;
document.getElementById(‘freightInYes’).classList.add(‘active’);
document.getElementById(‘freightInNo’).classList.remove(‘active’);
document.getElementById(‘pickupSection’).classList.remove(‘hidden’);
});

document.getElementById(‘freightInNo’).addEventListener(‘click’, () => {
quoteData.freightIn = false;
document.getElementById(‘freightInNo’).classList.add(‘active’);
document.getElementById(‘freightInYes’).classList.remove(‘active’);
document.getElementById(‘pickupSection’).classList.add(‘hidden’);
quoteData.pickupAddress = ”;
quoteData.pickupForklift = null;
updateStep(3);
countdownInterval1 = startCountdown(‘countdownMessage1’, 5, 4);
});

document.getElementById(‘pickupAddress’).addEventListener(‘input’, (e) => {
quoteData.pickupAddress = e.target.value;
});

function checkPickupComplete() {
if (quoteData.pickupAddress && quoteData.pickupForklift !== null) {
updateStep(3);
countdownInterval1 = startCountdown(‘countdownMessage1’, 5, 4);
}
}

document.getElementById(‘pickupForkliftYes’).addEventListener(‘click’, () => {
quoteData.pickupForklift = true;
document.getElementById(‘pickupForkliftYes’).classList.add(‘active’);
document.getElementById(‘pickupForkliftNo’).classList.remove(‘active’);
checkPickupComplete();
});

document.getElementById(‘pickupForkliftNo’).addEventListener(‘click’, () => {
quoteData.pickupForklift = false;
document.getElementById(‘pickupForkliftNo’).classList.add(‘active’);
document.getElementById(‘pickupForkliftYes’).classList.remove(‘active’);
checkPickupComplete();
});

// Freight Out
document.getElementById(‘freightOutYes’).addEventListener(‘click’, () => {
quoteData.freightOut = true;
document.getElementById(‘freightOutYes’).classList.add(‘active’);
document.getElementById(‘freightOutNo’).classList.remove(‘active’);
document.getElementById(‘deliverySection’).classList.remove(‘hidden’);
});

document.getElementById(‘freightOutNo’).addEventListener(‘click’, () => {
quoteData.freightOut = false;
document.getElementById(‘freightOutNo’).classList.add(‘active’);
document.getElementById(‘freightOutYes’).classList.remove(‘active’);
document.getElementById(‘deliverySection’).classList.add(‘hidden’);
quoteData.deliveryAddress = ”;
quoteData.deliveryForklift = null;
updateStep(5);
countdownInterval2 = startCountdown(‘countdownMessage2’, 5, 6, startLoadingSequence);
});

document.getElementById(‘deliveryAddress’).addEventListener(‘input’, (e) => {
quoteData.deliveryAddress = e.target.value;
});

function checkDeliveryComplete() {
if (quoteData.deliveryAddress && quoteData.deliveryForklift !== null) {
updateStep(5);
countdownInterval2 = startCountdown(‘countdownMessage2’, 5, 6, startLoadingSequence);
}
}

document.getElementById(‘deliveryForkliftYes’).addEventListener(‘click’, () => {
quoteData.deliveryForklift = true;
document.getElementById(‘deliveryForkliftYes’).classList.add(‘active’);
document.getElementById(‘deliveryForkliftNo’).classList.remove(‘active’);
checkDeliveryComplete();
});

document.getElementById(‘deliveryForkliftNo’).addEventListener(‘click’, () => {
quoteData.deliveryForklift = false;
document.getElementById(‘deliveryForkliftNo’).classList.add(‘active’);
document.getElementById(‘deliveryForkliftYes’).classList.remove(‘active’);
checkDeliveryComplete();
});

// Loading sequence
function startLoadingSequence() {
updateStep(6);
document.getElementById(‘loadingScreen’).classList.remove(‘hidden’);
document.getElementById(‘quoteSection’).classList.add(‘hidden’);

const loadingText = document.getElementById(‘loadingText’);
const loadingSubtext = document.getElementById(‘loadingSubtext’);

setTimeout(() => {
loadingText.textContent = ‘Analysing requirements…’;
loadingSubtext.textContent = ‘Checking carrier availability’;
}, 2000);

setTimeout(() => {
loadingText.textContent = ‘Calculating best routes…’;
loadingSubtext.textContent = ‘Optimizing for cost and time’;
}, 4000);

setTimeout(() => {
loadingText.textContent = ‘Applying surcharges…’;
loadingSubtext.textContent = ‘Tailgate fees, fuel levy’;
}, 6000);

setTimeout(() => {
loadingText.textContent = ‘Finalizing quote…’;
loadingSubtext.textContent = ‘Almost there’;
}, 8000);

setTimeout(() => {
document.getElementById(‘loadingScreen’).classList.add(‘hidden’);
document.getElementById(‘quoteSection’).classList.remove(‘hidden’);

// Update quote with data
if (quoteData.show) {
document.getElementById(‘quoteShow’).textContent = quoteData.show.name;
document.getElementById(‘quoteVenue’).textContent = quoteData.show.venue;
document.getElementById(‘quoteDates’).textContent = quoteData.show.dates;
}

// Calculate surcharges
let total = 1735;
let surcharge = 0;
if (quoteData.pickupForklift === false) surcharge += 65;
if (quoteData.deliveryForklift === false) surcharge += 65;

document.getElementById(‘quoteFreightIn’).textContent = quoteData.freightIn ? ‘$450’ : ‘$0’;
document.getElementById(‘quoteFreightOut’).textContent = quoteData.freightOut ? ‘$450’ : ‘$0’;
document.getElementById(‘quoteSurcharge’).textContent = `$${surcharge}`;
document.getElementById(‘quoteTotal’).textContent = `$${total + surcharge}`;
}, 10000);
}

// Accept quote
document.getElementById(‘acceptQuoteBtn’).addEventListener(‘click’, () => {
alert(‘Quote accepted! Thank you for choosing Axis Events.’);
});

// Initialize
updateStep(1);
</script>
